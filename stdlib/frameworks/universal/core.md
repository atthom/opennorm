## Axioms

All axioms in this framework (shown as Lean 4 code):

```lean4
-- ══════════════════════════════════════════════════════════════════════════
-- SEMANTIC QUANTIFIER DEFINITIONS
-- ══════════════════════════════════════════════════════════════════════════

-- Quantifier as a semantic type (predicate over LegalEntities)
-- Note: Quantifier is NOT generated by taxonomy - it's defined here as a semantic type
def Quantifier : Type := LegalEntities → Prop

-- Only wraps a specific entity - the entity must be carried through
def Only (e : LegalEntities) : Quantifier :=
  fun p => p = e

-- Universal quantifiers as semantic predicates
def AnyOne : Quantifier := fun _ => True
def NoOne : Quantifier := fun _ => False

-- ══════════════════════════════════════════════════════════════════════════
-- AXIOMS
-- ══════════════════════════════════════════════════════════════════════════

-- The primitive normative predicate
-- holds H actor action object q means:
--   position H holds for actor performing action on object
--   against persons satisfying quantifier q
axiom holds : Position → Quantifier → Action → Object → Quantifier → Prop

-- ══════════════════════════════════════════════════════════════════════════
-- OPERATORS (Z₂³ generators)
-- ══════════════════════════════════════════════════════════════════════════

-- C: Perspective automorphism (Holder ↔ Counterparty)
def C : Perspective → Perspective
  | .Holder => .Counterparty
  | .Counterparty => .Holder

-- O: Polarity automorphism (Positive ↔ Negative)
def O : Polarity → Polarity
  | .Positive => .Negative
  | .Negative => .Positive

-- E: Order automorphism (First ↔ Second)
def E : Order → Order
  | .First => .Second
  | .Second => .First

-- Involution proofs: each operator is self-inverse
theorem C_involution : ∀ x, C (C x) = x := by
  intro x
  cases x <;> rfl

theorem O_involution : ∀ x, O (O x) = x := by
  intro x
  cases x <;> rfl

theorem E_involution : ∀ x, E (E x) = x := by
  intro x
  cases x <;> rfl

-- Bilateral symmetry: flipping perspective swaps actor and counterparty
-- This axiom captures the fundamental symmetry of bilateral normative relations
-- When you view a normative relation from the counterparty's perspective (C),
-- the actor becomes the counterparty and the counterparty becomes the actor
axiom C_holds : 
  ∀ (persp : Perspective) (pol : Polarity) (ord : Order) 
    (actor : Quantifier) (action : Action) (object : Object) (q : Quantifier),
  holds (persp, pol, ord) actor action object q ↔
  holds (C persp, pol, ord) q action object actor
```

```lean4

-- Set type for subset relations
def Set (α : Type) : Type := α → Prop

def Set.subset {α : Type} (s1 s2 : Set α) : Prop :=
  ∀ x, s1 x → s2 x

instance {α : Type} : HasSubset (Set α) where
  Subset := Set.subset

-- ══════════════════════════════════════════════════════════════════════════
-- TYPE COERCIONS (LegalEntities → Quantifier)
-- ══════════════════════════════════════════════════════════════════════════

-- LegalEntity can be coerced to a Quantifier via the Only function
-- This allows legal entities to be used directly as actors in holds predicates
-- Example: Corporation "Acme Inc." can be used where Quantifier is expected
instance : Coe LegalEntities Quantifier := ⟨Only⟩

-- CollectiveBody has members that can be selected as a quantifier
-- This allows collective bodies to have dual nature:
--   1. As a single actor (via Coe LegalEntities above)
--   2. As a selector of their members (via HasMembers below)
-- Example: TreatyBody can act as single entity or select its member states
-- Note: Instances of HasMembers must be provided for specific collective bodies
class HasMembers (c : CollectiveBody) where
  members : Quantifier

-- ══════════════════════════════════════════════════════════════════════════
-- HOHFELDIAN POSITIONS (Z₂³ torsor)
-- ══════════════════════════════════════════════════════════════════════════

-- Note: Type aliases (Perspective, Polarity, Order) are auto-generated
-- from the taxonomy definitions above

-- Position encoded as (perspective, polarity, order)
abbrev Position := Perspective × Polarity × Order

-- Open namespaces for clean constructor access
open HohfeldianTaxonomy_Operator_Perspective (Holder Counterparty)
open HohfeldianTaxonomy_Operator_Polarity (Positive Negative)
open HohfeldianTaxonomy_Operator_Order (First Second)

-- The eight Hohfeldian positions
def Right      : Position := (Holder, Positive, First)
def Duty       : Position := (Counterparty, Positive, First)
def NoRight    : Position := (Holder, Negative, First)
def Privilege  : Position := (Counterparty, Negative, First)
def Power      : Position := (Holder, Positive, Second)
def Liability  : Position := (Counterparty, Positive, Second)
def Disability : Position := (Holder, Negative, Second)
def Immunity   : Position := (Counterparty, Negative, Second)

-- ══════════════════════════════════════════════════════════════════════════
-- CONDITIONAL NORMATIVE PREDICATES
-- ══════════════════════════════════════════════════════════════════════════

-- A Rule is a conditional normative position: it holds only when conditions are met
-- Multiple conditions can be combined into a single Prop using logical connectives (∧, ∨, etc.)
-- This captures defeasibility: normative rules that apply subject to exceptions or prerequisites
def Rule (condition : Prop) (H : Position) (actor : Quantifier) 
         (action : Action) (object : Object) (q : Quantifier) : Prop :=
  condition → holds H actor action object q

-- C operates only on the Hohfeldian structure, not on conditions
-- The bilateral symmetry (actor/counterparty swap) is independent of the conditions
theorem C_Rule :
  ∀ (cond : Prop) (persp : Perspective) (pol : Polarity) (ord : Order)
    (actor : Quantifier) (action : Action) (object : Object) (q : Quantifier),
  Rule cond (persp, pol, ord) actor action object q ↔
  Rule cond (C persp, pol, ord) q action object actor := by
  intro cond persp pol ord actor action object q
  unfold Rule
  constructor
  · intro h c
    exact (C_holds persp pol ord actor action object q).mp (h c)
  · intro h c
    exact (C_holds persp pol ord actor action object q).mpr (h c)

-- ══════════════════════════════════════════════════════════════════════════
-- METRICS
-- ══════════════════════════════════════════════════════════════════════════

-- Hamming distance: structural distance between positions
def hamming_distance (p q : Position) : Nat :=
  (if p.fst != q.fst then 1 else 0) +
  (if p.snd.fst != q.snd.fst then 1 else 0) +
  (if p.snd.snd != q.snd.snd then 1 else 0)

-- Alexandrov ordering: liberty ordering on positions
-- Privilege and Immunity have maximal liberty (value 3)
-- Right and Power have medium liberty (value 2)
-- Duty and Liability have minimal liberty (value 1)
-- NoRight and Disability have no liberty (value 0)
def liberty_value (p : Position) : Nat :=
  if p = Privilege ∨ p = Immunity then 3
  else if p = Right ∨ p = Power then 2
  else if p = Duty ∨ p = Liability then 1
  else 0

def alexandrov_leq (p q : Position) : Prop :=
  liberty_value p ≤ liberty_value q
```
